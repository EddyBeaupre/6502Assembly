CRC32_Check 

;---------------------------------------------
;CRC32_check
; Purposes:
;	1. Perform the CRC32 Checking
; Input: 
; Output:
; Required Variables:
; Required Functions:
; Required Table:
;	1. CRCT0
;	2. CRCT1
;	3. CRCT2
;	4. CRCT3
; Labels used:
;	1. CRC32_check:
; Variables used:
;	1. CRC...CRC+3
;	2. crc_len
;	3. tmp3:tmp2
; Constant used:
; Function used:
; Used by:
; Version: 1.0
; Revision Date: 28/8/2006
;---------------------------------------------
CRC32_check:
	LDY #$FF
	STY CRC
	STY CRC+1
	STY CRC+2
	STY CRC+3
	INY
LOOP	LDA (tmp2)
	INC tmp2
	EOR CRC         ; Quick CRC computation with lookup tables
	TAX
	LDA CRC+1
	EOR CRCT0,X
	STA CRC
	LDA CRC+2
	EOR CRCT1,X
	STA CRC+1
	LDA CRC+3
	EOR CRCT2,X
	STA CRC+2
	LDA CRCT3,X
	STA CRC+3
	INY
	CPY crc_len
	BNE LOOP
	LDY #3
COMPL	LDA CRC,Y
	EOR #$FF
	STA CRC,Y
	DEY
	BPL COMPL
	;compare the CRC register
	LDA (tmp2)
	CMP CRC+3
	BNE MSG_INVALID
	INC tmp2
	LDA (tmp2)
	CMP CRC+2
	BNE MSG_INVALID
	INC tmp2
	LDA (tmp2)
	CMP CRC+1
	BNE MSG_INVALID
	INC tmp2
	LDA (tmp2)
	CMP CRC
	BNE MSG_INVALID
MSG_VALID:
	; Correct Z=0
MSG_INVALID:
	; Incorrect Z<>0
	RTS    
;---------------------------------------------
