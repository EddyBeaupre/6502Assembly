I2C_Write1Byte2 

;---------------------------------------------
;I2C_Write1Byte2
; Purposes:
;	1. Send 1 Byte to FRAM.
; Input: 
; Output:
; Required Variables:
; Required Functions:
; Required Table:
; Labels used:
;	1. I2C_Write1Byte2:
; Variables used:
; Constant used:
; Function used:
; Used by:
; Version: 1.0
; Revision Date: 28/8/2006
;---------------------------------------------
I2C_Write1Byte2:
	;send 1st bit
	LDY #$00

	ASL A	;Rotate Adder
	BCC Send0_00	;Send 0
	LDY #%00000001	;Send 1
Send0_00:
	STY P3
	;trigger clock
	LDY #%00000010
	STY P2
	LDY #%00000000
	STY P2

	;send 2nd bit
	ASL A	;Rotate Adder
	BCC Send0_01	;Send 0
	LDY #%00000001	;Send 1
Send0_01:
	STY P3
	;trigger clock
	LDY #%00000010
	STY P2
	LDY #%00000000
	STY P2


	;send 3rd bit
	ASL A	;Rotate Adder
	BCC Send0_02	;Send 0
	LDY #%00000001	;Send 1
Send0_02:
	STY P3
	;trigger clock
	LDY #%00000010
	STY P2
	LDY #%00000000
	STY P2

	;send 4th bit
	ASL A	;Rotate Adder
	BCC Send0_03	;Send 0
	LDY #%00000001	;Send 1
Send0_03:
	STY P3
	;trigger clock
	LDY #%00000010
	STY P2
	LDY #%00000000
	STY P2

	;send 5th bit
	ASL A	;Rotate Adder
	BCC Send0_04	;Send 0
	LDY #%00000001	;Send 1
Send0_04:
	STY P3
	;trigger clock
	LDY #%00000010
	STY P2
	LDY #%00000000
	STY P2

	;send 6th bit
	ASL A	;Rotate Adder
	BCC Send0_05	;Send 0
	LDY #%00000001	;Send 1
Send0_05:
	STY P3
	;trigger clock
	LDY #%00000010
	STY P2
	LDY #%00000000
	STY P2

	;send 7th bit
	ASL A	;Rotate Adder
	BCC Send0_06	;Send 0
	LDY #%00000001	;Send 1
Send0_06:
	STY P3
	;trigger clock
	LDY #%00000010
	STY P2
	LDY #%00000000
	STY P2

	;send 8th bit
	ASL A	;Rotate Adder
	BCC Send0_07	;Send 0
	LDY #%00000001	;Send 1
Send0_07:
	STY P3
	;trigger clock
	LDY #%00000010
	STY P2
	LDY #%00000000
	STY P2

	RTS

;End of I2C_Write1Byte2
;---------------------------------------------
