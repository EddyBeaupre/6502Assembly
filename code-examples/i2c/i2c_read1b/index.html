I2C_Read1Byte 

;---------------------------------------------
;I2C_Read1Byte
; Purposes:
;	1. Read 1 Byte from FRAM.
; Input: 
; Output:
; Required Variables:
; Required Functions:
; Required Table:
; Labels used:
;	1. I2C_Read1Byte
; Variables used:
;	1. fram_data
; Constant used:
; Function used:
; Used by:
; Version: 1.0
; Revision Date: 28/8/2006
;---------------------------------------------
I2C_Read1Byte:
	STZ fram_data,X

	;rise clock
	LDA #%00000010
	STA P2

	;read data
	LDA P3	;read 1st bit
	AND #$01
	STZ P2	;set clock back to zero
	ASL A	;rotate adder

	STA fram_data,X

	;rise clock
	LDA #%00000010
	STA P2

	;read data
	LDA P3	;read 2nd bit
	AND #$01
	STZ P2	;set clock back to zero

	ORA fram_data,X
	ASL A	;rotate adder

	STA fram_data,X

	;rise clock
	LDA #%00000010
	STA P2

	;read data
	LDA P3	;read 3th bit
	AND #$01
	STZ P2	;set clock back to zero

	ORA fram_data,X
	ASL A	;rotate adder
	STA fram_data,X

	;rise clock
	LDA #%00000010
	STA P2

	;read data

	LDA P3	;read 4th bit

	AND #$01
	STZ P2	;set clock back to zero

	ORA fram_data,X
	ASL A	;rotate adder
	STA fram_data,X

	;rise clock
	LDA #%00000010
	STA P2

	;read data
	LDA P3	;read 5th bit
	AND #$01
	STZ P2	;set clock back to zero

	ORA fram_data,X
	ASL A	;rotate adder
	STA fram_data,X

	;rise clock
	LDA #%00000010
	STA P2

	;read data
	LDA P3	;read 6th bit
	AND #$01
	STZ P2	;set clock back to zero

	ORA fram_data,X
	ASL A	;rotate adder
	STA fram_data,X

	;rise clock
	LDA #%00000010
	STA P2

	;read data
	LDA P3	;read 7th bit
	AND #$01
	STZ P2	;set clock back to zero

	ORA fram_data,X
	ASL A	;rotate adder
	STA fram_data,X

	;rise clock
	LDA #%00000010
	STA P2

	;read data
	LDA P3	;read 8th bit
	AND #$01
	STZ P2	;set clock back to zero

	ORA fram_data,X
	STA fram_data,X

	RTS
;End of I2C_Read1Byte
;---------------------------------------------
