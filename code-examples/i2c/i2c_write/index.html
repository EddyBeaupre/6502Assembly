I2C_Write 

;---------------------------------------------
;I2C_Write
; Purposes:
;	1. Write Data to FRAM.
; Input: 
;	1. fram_page
;	2. fram_add
;	3. fram_data
;	4. fram_length
; Output:
; Required Variables:
; Required Functions:
; Required Table:
; Labels used:
;	1. I2C_Write:
;	2. FRAM_CON3:
;	3. FRAM_WRITELOOP:
; Variables used:
;	1. fram_page
;	2. fram_add
;	3. fram_data
;	4. fram_length
;	5. fram_sts
; Constant used:
; Function used:
;	1. I2C_Config
;	2. I2C_Write1Byte2
; Used by:
; Version: 1.0
; Revision Date: 28/8/2006
;---------------------------------------------
I2C_Write:
	JSR I2C_Config

	STZ p2	;Reset SCL
	;issue a stop Command
	STZ p3	;Reset SDA
	LDA #%00000010
	STA P2
	LDA #%00000001
	STA P3

	;issue a start Command
	LDA #%00000000
	STA P3

	;Reset SCL, ready for transmission
	LDA #%00000000
	STA P2

	;send the Slave ID, ie, 1010

	;send 1 to SDA
	LDA #%00000001
	STA P3

	;trigger clock
	LDA #%00000010
	STA P2
	LDA #%00000000
	STA P2

	;send 0 to SDA
	LDA #%00000000
	STA P3

	;trigger clock
	LDA #%00000010
	STA P2
	LDA #%00000000
	STA P2

	;send 1 to SDA
	LDA #%00000001
	STA P3

	;trigger clock
	LDA #%00000010
	STA P2
	LDA #%00000000
	STA P2

	;send 0 to SDA
	LDA #%00000000
	STA P3

	;trigger clock
	LDA #%00000010
	STA P2
	LDA #%00000000
	STA P2

	;send the Device select, 00
	;send 0 to SDA
	LDA #%00000000
	STA P3

	;trigger clock
	LDA #%00000010
	STA P2
	LDA #%00000000
	STA P2

	;send 0 to SDA
	LDA #%00000000
	STA P3

	;trigger clock
	LDA #%00000010
	STA P2
	LDA #%00000000
	STA P2

	;send the page select
	LDA fram_page
	STA P3
	;trigger clock
	LDA #%00000010
	STA P2
	LDA #%00000000
	STA P2

	;send 0 to SDA
	;writing data
	LDA #%00000000
	STA P3

	;trigger clock
	LDA #%00000010
	STA P2
	LDA #%00000000
	STA P2

	;Acknowledgement
	;Port 3.0/Port 3.1 are both set as input pins
	STZ p3_dir

	LDA #%00000010
	STA P2
	LDA p3
	AND #%00000001
	BEQ FRAM_CON3
	RTS

FRAM_CON3:
	LDA #%00000000
	STA P2

	LDA #%00000001
	STA p3_dir

	LDA fram_add
	JSR I2C_Write1Byte2

FRAM_WRITELOOP:
	;Acknowledgement
	;Port 3.0/Port 3.1 are both set as input pins
	STZ p3_dir

	LDA #%00000010
	STA P2
	LDA p3
	AND #%00000001
	LDA #%00000000
	STA P2

	LDA #%00000001
	STA p3_dir

	LDA fram_data,X
	JSR I2C_Write1Byte2

	INX
	CPX fram_length
	BNE FRAM_WRITELOOP
	
	;Acknowledgement
	;Port 3.0/Port 3.1 are both set as input pins
	STZ p3_dir

	LDA #%00000010
	STA P2
	LDA p3
	AND #%00000001
	LDA #%00000000
	STA P2

	;Port 3.0/Port 3.1 are both set as output pins
	LDA #%00000001
	STA p3_dir

	STZ p3	;Reset SDA
	LDA #%00000010
	STA P2
	LDA #%00000001
	STA P3

	STZ fram_sts

	RTS
;End of I2C_Write
;---------------------------------------------
